+            />
+          </div>
+          <div className="space-y-2">
+            <Label htmlFor="title">Title</Label>

+                        {appt.title}
+                      </h3>
+                      <p className="text-sm text-muted-foreground">Student: {appt.studentName}</p>

+                          {appt.type}

+                        <span className="text-xs text-muted-foreground">
